NCardsABWin

题目描述

面值为1～10的牌组成一组，每次你从组里等概率的抽出1～10中的一张，下次抽会换一个新的组，有无限组，当累加和<17时，你将一直抽牌，当累加和>=17且<21时，你将获胜，当累加和>=21时，你将失败，返回获胜的概率。

题解

设p(sum)为当前累加和为sum时获胜的概率，显然，p(17)=p(18)=p(19)=p(20)=1, p(21)=p(22)=…=0;

所以当你来到cur这个累加和的时候，获胜的概率为 （p(cur + 1) + p(cur + 2) + …… + p(cur+10)) / 10;

代码

public class NCardsABWin {
    //递归解法
    public static double recursion(){
        return process(0);
    }

    public static double process(int sum){
        if (sum >= 17 && sum < 21) {
            return 1.0;
        }
        if (sum >= 21) {
            return 0.0;
        }
        double ans = 0.0;
        for (int i = 1; i <= 10; i++) {
            ans += process(sum + i);
        }

        return ans / 10;
    }

    //非递归解法
    public static double no_recursion() {
        double[] p = new double[27];
        for (int i = 17; i < 21; i++) {
            p[i] = 1.0;
        }
        for (int i = 21; i < 27; i++) {
            p[i] = 0.0;
        }
        for (int i = 16; i >= 0; i--) {
            for (int c = 1; c <= 10; c++) {
                p[i] += p[i + c];
            }
            p[i] /= 10;
        }
        return p[0];
    }

    public static void main(String[] args) {
        System.out.println(recursion());
        System.out.println(no_recursion());
    }

}

扩展题：

题目描述

面值为1～N的牌组成一组，每次你从组里等概率的抽出1～N中的一张，下次抽会换一个新的组，有无限组，当累加和<a时，你将一直抽牌，当累加和>=a且<b时，你将获胜，当累加和>=b时，你将失败，返回获胜的概率。给定的参数为N，a，b

题解

首先排除几个参数错误的情况，N<1没有牌，a>=b区间错误，b<=0无法获胜，直接返回0；还有一种特殊情况：b - a >= N 直接返回1（由基本题的非递归解法可得到）

然后同基本题思路一样，当你来到cur这个累加和的时候，获胜的概率为 （p(cur + 1) + p(cur + 2) + …… + p(cur+N)) / N;同样可以采用递归的方法，但当参数范围过大时，需要进行优化，此题可以用到观察位置优化枚举的技巧，或者改记忆化搜索或动态规划。

代码

import java.math.BigDecimal;

public class NCardsABWin {
    //递归解法
    public static double recursion1(){
        return p(0);
    }

    // 当你来到sum这个累加和的时候，获胜概率是多少返回！
    public static double p(int sum){
        if (sum >= 17 && sum < 21) {
            return 1.0;
        }
        if (sum >= 21) {
            return 0.0;
        }
        double ans = 0.0;
        for (int i = 1; i <= 10; i++) {
            ans += p(sum + i);
        }

        return ans / 10;
    }

    //非递归解法
    public static double no_recursion1() {
        double[] p = new double[27];
        for (int i = 17; i < 21; i++) {
            p[i] = 1.0;
        }
        for (int i = 21; i < 27; i++) {
            p[i] = 0.0;
        }
        for (int i = 16; i >= 0; i--) {
            for (int c = 1; c <= 10; c++) {
                p[i] += p[i + c];
            }
            p[i] /= 10;
        }
        return p[0];
    }


    //递归解法
    public static double recursion2(int N, int a, int b) {
        if (N < 1 || a >= b || b <= 0) {
            return 0.0;
        }
        if (b - a >= N) {
            return 1.0;
        }
        return p2(N, a, b, 0);
    }

    public static double p2(int N, int a, int b, int sum){
        if (sum >= a && sum < b) {
            return 1.0;
        }
        if (sum >= b) {
            return 0.0;
        }
        double ans = 0.0;
        for (int i = 1; i <= N; i++) {
            ans += p2(N, a, b, sum + i);
        }
        return ans / N;
    }

    //递归优化，用到观察位置优化枚举的技巧
    public static double recursion3(int N, int a, int b) {
        if (N < 1 || a >= b || b <= 0) {
            return 0.0;
        }
        if (b - a >= N) {
            return 1.0;
        }
        return p3(N, a, b, 0);
    }


    public static double p3(int N, int a, int b, int sum){
        if (sum >= a && sum < b) {
            return 1.0;
        }
        if (sum >= b) {
            return 0.0;
        }
        double ans = 0.0;
//        for (int i = 1; i <= N; i++) {
//            ans += p2(N, a, b, sum + i);
//        }
//      p(cur) = (p(cur + 1) + p(cur + 2) + ... + p(cur + N)) / N
//    p(cur+1) = (p(cur + 2) + p(cur + 3) + ... + p(cur+N+1)) / N
//    p(cur+1)*N - p(cur+N+1) = p(cur + 2) + p(cur + 3) + ... + p(cur + N)
//      p(cur) = (p(cur+1) + p(cur+1)*N - p(cur+N+1)) / N
//      但是p(a) = 1 != p(a+1) + p(a+2) + ... + p(a+N)/N
//      所以 p(a-1) 不能用上述公式求，即 p(a-1) 为特殊情况， p(a-1) = (b - a) / N
//      当sum<a-1 p(sum) = (p(sum+1) + p(sum+1)*N - p(sum+N+1)) / N;
        if (sum == a - 1) {
            return 1.0 * (b - a) / N;     //细节注意*1.0转化为double
        }
        ans = p3(N, a, b, sum + 1) + p3(N, a, b, sum + 1) * N - p3(N, a, b, sum + N + 1);
        return ans / N;
    }

    //    动态规划
    public static double dp(int N, int a, int b) {
        if (N < 1 || a >= b || b <= 0) {
            return 0.0;
        }
        if (b - a >= N) {
            return 1.0;
        }
        double[] p = new double[b];
        for (int i = a; i < b; i++) {
            p[i] = 1.0;
        }
        if (a - 1 >= 0) {
            p[a - 1] = 1.0*(b - a) / N;
        }

        for (int cur = a - 2; cur >= 0; cur--) {
            double ans = p[cur + 1] * N + p[cur + 1];
            if (cur + N + 1 < b) {
                ans -= p[cur + N + 1];
            }
            p[cur] = ans / N;
        }

        return p[0];
    }

    public static void main(String[] args) {
        int N = 10;
        int a = 17;
        int b = 21;
        System.out.println("N = " + N + ", a = " + a + ", b = " + b);
        System.out.println(recursion1());
        System.out.println(no_recursion1());
        System.out.println(recursion2(N, a, b));
        System.out.println(recursion3(N, a, b));
        System.out.println(dp(N, a, b));



        int maxN = 15;
        int maxM = 20;
        int times = 100000;
        System.out.println("测试开始");
        System.out.print("比对double类型答案可能会有精度对不准的问题, ");
        System.out.print("所以答案一律只保留小数点后四位进行比对, ");
        System.out.println("如果没有错误提示, 说明验证通过");
        for (int i = 0; i < times; i++) {
            N = (int) (Math.random() * maxN);
            a = (int) (Math.random() * maxM);
            b = (int) (Math.random() * maxM);
            double ans2 = new BigDecimal(recursion2(N, a, b)).setScale(4, BigDecimal.ROUND_HALF_UP).doubleValue();
            double ans3 = new BigDecimal(recursion3(N, a, b)).setScale(4, BigDecimal.ROUND_HALF_UP).doubleValue();
            double ans4 = new BigDecimal(dp(N, a, b)).setScale(4, BigDecimal.ROUND_HALF_UP).doubleValue();
            if (ans2 != ans3 || ans2 != ans4) {
                System.out.println("Oops!");
                System.out.println(N + " , " + a + " , " + b);
                System.out.println(ans2);
                System.out.println(ans3);
                System.out.println(ans4);
            }

        }
        System.out.println("测试结束");

        N = 10000;
        a = 67834;
        b = 72315;
        System.out.println("N = " + N + ", a = " + a + ", b = " + b + "时, 除了动态规划方法外都超时");
        System.out.print("dp答案: ");
        System.out.println(dp(N, a, b));

    }

}
